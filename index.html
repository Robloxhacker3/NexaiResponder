import requests
import urllib.parse

class NexaiResponder:
    def __init__(self):
        self.base_url = "https://nexairesponder.vercel.app/api/"
    
    def get_response(self, message):
        """Send a message and get AI response"""
        # URL encode the message
        encoded_message = urllib.parse.quote(message)
        url = f"{self.base_url}message={encoded_message}"
        
        try:
            response = requests.get(url, timeout=10)
            response.raise_for_status()
            return response.json()
        except requests.exceptions.RequestException as e:
            return {"error": f"Request failed: {str(e)}"}
    
    def chat(self, message):
        """Simple chat interface"""
        result = self.get_response(message)
        if "error" not in result:
            return result["respond"]
        return result["error"]

class NexaiResponder {
    constructor() {
        this.baseUrl = 'https://nexairesponder.vercel.app/api/';
    }

    async getResponse(message) {
        try {
            // URL encode the message
            const encodedMessage = encodeURIComponent(message);
            const url = `${this.baseUrl}message=${encodedMessage}`;
            
            const response = await fetch(url);
            if (!response.ok) {
                throw new Error(`HTTP error! status: ${response.status}`);
            }
            
            return await response.json();
        } catch (error) {
            return { error: `Request failed: ${error.message}` };
        }
    }

    async chat(message) {
        const result = await this.getResponse(message);
        return result.error ? result.error : result.respond;
    }
}

// React Component Example
function ChatComponent() {
    const [message, setMessage] = React.useState('');
    const [response, setResponse] = React.useState('');
    const [loading, setLoading] = React.useState(false);

    const sendMessage = async () => {
        if (!message.trim()) return;
        
        setLoading(true);
        const ai = new NexaiResponder();
        const result = await ai.getResponse(message);
        
        setResponse(result.respond || result.error || 'No response');
        setLoading(false);
    };

    return React.createElement('div', { className: 'chat-container' }, [
        React.createElement('input', {
            key: 'input',
            type: 'text',
            value: message,
            onChange: (e) => setMessage(e.target.value),
            placeholder: 'Type your message...',
            onKeyPress: (e) => e.key === 'Enter' && sendMessage()
        }),
        React.createElement('button', {
            key: 'button',
            onClick: sendMessage,
            disabled: loading
        }, loading ? 'Sending...' : 'Send'),
        React.createElement('div', {
            key: 'response',
            className: 'response'
        }, response)
    ]);
}

const express = require('express');
const axios = require('axios');
const app = express();

app.use(express.json());

class NexaiResponder {
    constructor() {
        this.baseUrl = 'https://nexairesponder.vercel.app/api/';
    }

    async getResponse(message) {
        try {
            const encodedMessage = encodeURIComponent(message);
            const url = `${this.baseUrl}message=${encodedMessage}`;
            
            const response = await axios.get(url, { timeout: 10000 });
            return response.data;
        } catch (error) {
            return { 
                error: `Request failed: ${error.response?.data?.error || error.message}` 
            };
        }
    }

    async chat(message) {
        const result = await this.getResponse(message);
        return result.error ? result.error : result.respond;
    }
}

// Batch processing endpoint
app.post('/batch', async (req, res) => {
    try {
        const { messages } = req.body;
        
        if (!Array.isArray(messages)) {
            return res.status(400).json({ error: 'Messages must be an array' });
        }

        const responses = await Promise.all(
            messages.map(async (message, index) => {
                const response = await ai.getResponse(message);
                return {
                    index,
                    message,
                    response: response.respond || response.error,
                    success: !response.error
                };
            })
        );

        res.json({ responses, total: messages.length });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
});
